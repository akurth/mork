59      @Test
60      public void concat() {
61          long left;
62          Prefix right;
63          long test;
64  
65          left = prefix(10, 11).data;
66          right = prefix(20, 21, 22);
67          test = Prefix.concat(left, right.data, 2);
68          assertEquals(left, test);
69          check(Prefix.concat(left, right.data, 3), 10, 11, 20);
70          check(Prefix.concat(left, right.data, 4), 10, 11, 20, 21);
71          check(Prefix.concat(left, right.data, 5), 10, 11, 20, 21, 22);
72          check(Prefix.concat(left, right.data, 6), 10, 11, 20, 21, 22);
73          check(Prefix.concat(left, right.data, 7), 10, 11, 20, 21, 22);
74      }
75  
76      private void check(long actual, int ... expected) {
77          assertTrue(Arrays.equals(Prefix.unpack(actual), expected));
78      }
79  
80      private Prefix prefix(int ... symbols) {
81          PrefixSet set;
82          Prefix result;
83  
84          set = new PrefixSet();
85          set.addUnpacked(symbols);
86          result = set.iterator();
87          result.step();
88          return result;
89      }

99          hash = Prefix.hashFirst(prefix, hit.length);
