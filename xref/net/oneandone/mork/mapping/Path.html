32   * Visibility of some Definition, kind of an Argument builder.
33   */
34  public class Path {
35      public static final int DOWN = 0;
36      public static final int DOWNS = 1;
37      public static final int UP = 2;
38      public static final int UPS = 3;

40      /** stopper: yes; children: yes. */
41      public static final int ISOLATED = 0;
42  
43      /** alternatives get merged. */
44      public static final int MERGEABLE = 1;

47       * @param targets   list of Definitions
48       */
49      public static void translate(Syntax syntax,
50                Definition source, int move, IntBitSet stopper, List<Definition> targets, int modifier) throws GenericException {
51          translate(syntax, modifier, source, targets,
52                    new int[] { move }, new IntBitSet[] { new IntBitSet(stopper) });
53      }

55      /**
56       * Creates a path with no steps
57       */
58      public static void translate(Syntax syntax, Definition source, Definition target) throws GenericException {
59          List<Definition> targets;
60          IntBitSet stopper;
61  
62          targets = new ArrayList<Definition>();
63          targets.add(target);
64          stopper = new IntBitSet();
65          stopper.add(target.getAttribute().symbol);
66          translate(syntax, ISOLATED, source, targets, new int[] {}, new IntBitSet[] { stopper });
67      }

69      /**
70       * Creates a path with 1+ steps
71       */
72      public static void translate(Syntax syntax,
73              Definition source, int[] moves, int[] symbols, Definition target, int modifier) throws GenericException {
74          IntBitSet[] stoppers;
75          List<Definition> targets;
76          int i;
77  
78          if (moves.length - 1 != symbols.length) {
79              throw new IllegalArgumentException();
80          }
81  
82          stoppers = new IntBitSet[moves.length];
83          targets = new ArrayList<Definition>();
84          targets.add(target);
85          for (i = 0; i < symbols.length; i++) {
86              stoppers[i] = new IntBitSet();
87              stoppers[i].add(symbols[i]);
88          }
89          stoppers[i] = new IntBitSet();
90          stoppers[i].add(target.getAttribute().symbol);
91  
92          translate(syntax, modifier, source, targets, moves, stoppers);
93      }

95      /**
96       * The method actually doing the work
97       */
98      private static void translate(Syntax syntax,
99              int modifier, Definition source, List<Definition> targets, int[] moves, IntBitSet[] stoppers) throws GenericException {
100         Path path;
101         int count;
102 
103         path = new Path(syntax.getGrammar(), modifier, source, targets, moves, stoppers);
104         count = path.translate();
105         if (count == 0 && source.getAttribute().symbol != syntax.getGrammar().getStart()) {
106             // TODO: improved error message
107             throw new GenericException("dead-end path for attribute " + source.getName());
108         }
109     }

114     private final Grammar grammar;
115 
116     private final Definition source;
117 
118     /** List of Definitions (in last stopper). */
119     private final List<Definition> targets;
120 
121     // moves.length == stoppers.length == copyBuffers.length
122     private final int[] moves;
123     private final IntBitSet[] stoppers;
124     private final int modifier;

128     private Path(Grammar grammar,
129                  int modifier, Definition source, List<Definition> targets, int[] moves, IntBitSet[] stoppers)
130     {
131         this.grammar = grammar;
132         this.modifier = modifier;
133         this.source = source;
134         this.targets = targets;
135         this.moves = moves;
136         this.stoppers = stoppers;
137         this.copyBuffers = new List[moves.length];
138     }

140     private int translate() throws GenericException {
141         int step;
142         int max;
143         List<AgBuffer> buffers;
144         Definition target;
145         Argument arg;
146         List<Definition> sources;
147 
148         buffers = new ArrayList<AgBuffer>();
149         buffers.add(new AgBuffer(source.getAttribute()));
150         for (step = 0; step < moves.length; step++) {
151             translateStep(step, buffers);
152             buffers = copyBuffers[step];
153         }
154         max = buffers.size();
155         for (step = 0; step < max; step++) {
156             sources = new ArrayList<Definition>();
157             sources.add(source);
158             arg = new Argument(modifier, buffers.get(step), sources);
159             target = lookupTarget(arg.getAttribute().symbol);
160             target.addArgument(arg, source);
161         }
162         return max;
163     }

165     private void translateStep(int step, List<AgBuffer> initialBuffers) {
166         int i;
167         int max;
168         AgBuffer buffer;
169         IntBitSet targetSymbols;
170 
171         if (step == moves.length - 1) {
172             targetSymbols = getTargetSymbols(targets);
173         } else {
174             targetSymbols = stoppers[step];
175         }
176 
177         copyBuffers[step] = new ArrayList<AgBuffer>();
178         max = initialBuffers.size();
179         for (i = 0; i < max; i++) {
180             buffer = initialBuffers.get(i);
181             prefixedTransport(step, buffer, stoppers[step], targetSymbols);
182         }
183     }
184 
185     private void prefixedTransport(
186         int step, AgBuffer prefixBuffer, IntBitSet border, IntBitSet targetSymbols)
187     {
188         int ofs;
189         int max;
190         int i;
191         Attribute oldAttr;
192         Attribute newAttr;
193         AgBuffer buffer;
194         AgBuffer tmp;
195         List<AgBuffer> resultBuffers;
196 
197         Attribute attr = prefixBuffer.getStart();
198         resultBuffers = copyBuffers[step];
199         ofs = resultBuffers.size();
200         transport(grammar, attr, moves[step], border, targetSymbols, resultBuffers);
201         max = resultBuffers.size();
202         for (i = ofs; i < max; i++) {
203             oldAttr = resultBuffers.get(i).getStart();
204             tmp = new AgBuffer((Attribute) null);
205             tmp.append(prefixBuffer);
206             tmp.append(resultBuffers.get(i));
207             buffer = new AgBuffer((Attribute) null);
208             newAttr = buffer.cloneAttributes(tmp, oldAttr.type, oldAttr);
209             buffer.setStart(newAttr);
210             resultBuffers.set(i, buffer);
211         }
212     }

214     private static IntBitSet getTargetSymbols(List<Definition> defs) {
215         IntBitSet result;
216 
217         result = new IntBitSet();
218         for (Definition def : defs) {
219             result.add(def.getAttribute().symbol);
220         }
221         return result;
222     }
223 
224     private Definition lookupTarget(int symbol) {
225         for (Definition def : targets) {
226             if (def.getAttribute().symbol == symbol) {
227                 return def;
228             }
229         }
230         return null;
231     }

235     private static void transport(Grammar grammar, Attribute seed, int move, IntBitSet rawBorder,
236                                   IntBitSet targetSymbols, List<AgBuffer> resultBuffers) {
237         AgBuffer commulated;
238         IntBitSet border;
239         boolean down;
240         List<Attribute> attrs;
241         int i;
242         int max;
243         Attribute dest;
244         AgBuffer tmp;
245         int card;
246         AgBuffer buffer;
247         Attribute attr;
248         Occurrence occ;
249 
250         if (move == Path.DOWN || move == Path.UP) {
251             border = new IntBitSet();
252             grammar.getSymbols(border);
253         } else {
254             border = rawBorder;
255         }
256         down = (move == Path.DOWN || move == Path.DOWNS);
257         if (down && !border.contains(seed.symbol)) {
258             // TODO: needed for \Block//VariableReference in compiler examples
259             border.add(seed.symbol);
260         }
261         commulated = Pusher.run(down, seed, border, grammar);
262         attrs = commulated.getTransportAttributes();
263         max = attrs.size();
264         for (i = 0; i < max; i++) {
265             dest = attrs.get(i);
266             if (dest != seed && targetSymbols.contains(dest.symbol)) {
267                 tmp = commulated.createReduced(dest);
268                 occ = null;
269                 if (down) {
270                     card = tmp.isDownOptional()? Type.OPTION : Type.VALUE;
271                     card = Type.cardCard(card, seed.type.card);
272                 } else {
273                     occ = tmp.calcOccurrence(dest);
274                     card = Type.cardCard(occ.card(), seed.type.card);
275                     if (seed.type.card == Type.SEQUENCE || occ.max == Occurrence.UNBOUNDED) {
276                         occ = null;  // don't split
277                     }
278                 }
279                 if (occ == null) {
280                     buffer = new AgBuffer((Attribute) null);
281                     attr = buffer.cloneAttributes(tmp, new Type(seed.type.type, card), dest);
282                     buffer.setStart(attr);
283                     resultBuffers.add(buffer);
284                 } else {
285                     createSplitted(tmp, seed.type.type, occ, dest, resultBuffers);
286                 }
287             }
288         }
289     }

291     private static void createSplitted(AgBuffer orig, Class<?> cls, Occurrence occ,
292                                        Attribute origDest, List<AgBuffer> resultBuffers) {
293         int seq;
294         AgBuffer tmp;
295         AgBuffer buffer;
296         Attribute attr;
297         Type type;
298         Attribute dest;
299 
300         orig.calcOccurrences();
301         for (seq = 0; seq < occ.max; seq++) {
302             tmp = new AgBuffer((Attribute) null);
303             dest = orig.createSequence(origDest, seq, tmp);
304 
305             if (seq < occ.min) {
306                 type = new Type(cls, Type.VALUE);
307             } else {
308                 type = new Type(cls, Type.OPTION);
309             }
310             buffer = new AgBuffer((Attribute) null);
311             attr = buffer.cloneAttributes(tmp, type, dest);
312             buffer.setStart(attr);
313             resultBuffers.add(buffer);
314         }
315     }
