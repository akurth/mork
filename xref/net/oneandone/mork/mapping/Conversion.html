35          "Argument type mismatch";

39      private static void throwIllegalCall(
40          String kind, Selection selection, Definition def, List<Argument> args) throws GenericException {
41          StringBuilder msg;
42          int max;
43          int i;
44  
45          msg = new StringBuilder();
46          msg.append("for attribute ");
47          msg.append(def.getName());
48          msg.append(":\n");
49          msg.append("  constructors:\n");
50          max = selection.size();
51          for (i = 0; i < max; i++) {
52              msg.append("    " + selection.getFunction(i) + "\n");
53          }
54          msg.append("  arguments:\n");
55          max = args.size();
56          for (Argument arg : args) {
57              msg.append("    " + arg.getAttribute().type + " " + arg.getSourcesString() + "\n");
58          }
59          throw new GenericException(kind, msg.toString());
60      }

62      public static Function find(Selection selection, Definition def, List<Argument> args, List<Attribute> outAttrs) throws GenericException {
63          int i;
64          int max;
65          Function tmp;
66          Function resultFn;
67          List<Attribute> resultArgs;
68          List<Attribute> tmpInArgs;
69          List<Attribute> tmpOutArgs;
70  
71          resultFn = null;
72          resultArgs = null;
73          max = selection.size();
74          for (i = 0; i < max; i++) {
75              tmpInArgs = getAttributes(args);
76              tmpOutArgs = new ArrayList<Attribute>();
77              tmp = arrange(selection.getFunction(i), tmpInArgs, tmpOutArgs);
78              if (tmp != null) {
79                  if (resultFn != null) {
80                      throwIllegalCall(AMBIGUOUS_CONSTRUCTOR_NAME, selection, def, args);
81                  }
82                  resultFn = tmp;
83                  resultArgs = tmpOutArgs;
84              }
85          }
86          if (resultFn == null) {
87              throwIllegalCall(ARGUMENT_TYPE_MISMATCH, selection, def, args);
88          }
89  
90          outAttrs.addAll(resultArgs);
91          return resultFn;
92      }

95          List<Attribute> lst;

97          lst = new ArrayList<Attribute>();
98          for (Argument arg : args) {
99              lst.add(arg.getAttribute());
100         }
101         return lst;

104     private static Function arrange(Function fn, List<Attribute> inArgs, List<Attribute> outArgs) {
105         int i;
106         int max;
107 
108         max = fn.getParameterTypes().length;
109         for (i = 0; i < max; i++) {
110             fn = arrangeArg(fn, i, inArgs, outArgs);
111             if (fn == null) {
112                 return null;
113             }
114         }
115         if (inArgs.size() == 0) {
116             return fn;
117         } else {
118             return null;
119         }
120     }

122     private static Function arrangeArg(Function fn, int idx, List<Attribute> inArgs, List<Attribute> outArgs) {
123         Class<?> type;
124         Attribute attr;
125 
126         type = fn.getParameterTypes()[idx];
127         attr = removeAssignable(type, inArgs);
128         if (attr == null) {
129             return null;
130         }
131         outArgs.add(attr);
132         if (type.isArray()) {
133             // List -> <reference>[]
134             //  or
135             // List -> <primitive>[]
136             return new Composition(fn, idx, new ToArray(type.getComponentType()));
137         } else if (List.class.isAssignableFrom(type)) {
138             // List -> List
139             // no conversion required:
140             return fn;
141         } else {
142             // <reference> -> <reference>
143             // <reference> -> <primitive>
144             if ((attr.type.card == Type.OPTION) || type.isPrimitive()) {
145                 // Option is used to unwrap into primitive arguments;
146                 // if not an OPTION, the optional test inside is never true.
147                 fn = new Option(fn, idx);
148             } else {
149                 // no conversion
150             }
151             return fn;
152         }
153     }

155     private static Attribute removeAssignable(Class<?> type, List<Attribute> inArgs) {
156         int i;
157         int max;
158         Attribute current;
159 
160         max = inArgs.size();
161         for (i = 0; i < max; i++) {
162             current = inArgs.get(i);
163             if (isAssignableFrom(type, current.type)) {
164                 inArgs.remove(i);
165                 return current;
166             }
167         }
168         return null;
169     }

171     public static boolean hasFormalArgument(Selection sel, Type type) {
172         int i;
173         int max;
174 
175         if (sel == null) {
176             // do nothing - internal constructor has no formal argumets
177         } else {
178             max = sel.size();
179             for (i = 0; i < max; i++) {
180                 if (hasFormalArgument(sel.getFunction(i), type)) {
181                     return true;
182                 }
183             }
184         }
185         return false;
186     }

188     public static boolean hasFormalArgument(Function fn, Type type) {
189         Class<?>[] formalTypes;
190         int i;
191 
192         formalTypes = fn.getParameterTypes();
193         for (i = 0; i < formalTypes.length; i++) {
194             if (isAssignableFrom(formalTypes[i], type)) {
195                 return true;
196             }
197         }
198         return false;
199     }

201     public static boolean isAssignableFrom(Class<?> formalType, Type actual) {
202         Class<?> compType;
203 
204         if (actual.card == Type.SEQUENCE) {
205             if (List.class.isAssignableFrom(formalType)) {
206                 return true;
207             }
208             if (!formalType.isArray()) {
209                 return false;
210             }
211             if (formalType.isAssignableFrom(actual.type)) {
212                 return true;
213             }
214             compType = formalType.getComponentType();
215         } else {
216             compType = formalType;
217         }
218         if (compType.isAssignableFrom(actual.type)) {
219             return true;
220         }
221         if (compType.isPrimitive()) {
222             if (ClassRef.wrappedType(compType).isAssignableFrom(actual.type)) {
223                 return true;
224             }
225         }
226         return false;
227     }
