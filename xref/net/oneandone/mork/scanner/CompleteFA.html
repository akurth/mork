16  package net.oneandone.mork.scanner;
17  
18  import net.oneandone.mork.regexpr.Range;
19  
20  /**
21   * Adds tranisitions to a specified state for all input not defined
22   * in the given automaton. Transitions are sorted and concatenated if
23   * possible.
24   */
25  
26  public class CompleteFA {
27      /**
28       * The last state of the automaton returned is the error state.
29       */
30      public static FA create(FA fa, int errorSi) {
31          FA cfa; // result
32          int[] sorted;
33          int si;
34          int idx;
35          int faSize;
36          State faState;
37          State cfaState;
38          int nextChar;
39          int width;
40          Range range;
41          int nextSi;
42  
43          faSize = fa.size();
44          cfa = new FA();
45          for (si = 0; si < faSize; si++) {
46              faState = fa.get(si);
47              if (cfa.add(faState.getLabel()) != si) {
48                  throw new RuntimeException();
49              }
50              cfaState = cfa.get(si);
51              if (fa.getStart() == si) {
52                  cfa.setStart(si);
53              }
54              if (fa.isEnd(si)) {
55                  cfa.setEnd(si);
56              }
57  
58              sorted = faState.sortRanges();
59              nextChar = 0;
60              for (idx = 0; idx < sorted.length; idx += width) {
61                  width = countConsecutive(faState, sorted, idx);
62                  range = firstToLast(faState, sorted, idx, width);
63                  nextSi = faState.getEnd(sorted[idx]);
64                  if (nextChar < range.getFirst()) {
65                      cfaState.add(errorSi, new Range((char) nextChar, (char) (range.getFirst() - 1)));
66                  }
67                  cfaState.add(faState.getEnd(sorted[idx]), range);
68                  nextChar = range.getLast() + 1;
69              }
70              if (nextChar <= Character.MAX_VALUE) {
71                  cfaState.add(errorSi, new Range((char) nextChar, Character.MAX_VALUE));
72              }
73          }
74  
75          return cfa;
76      }
77  
78      private static int countConsecutive(State state, int[] tis, int ofs) {
79          int i;
80          int endSi;
81          int nextChar;
82          Range tmp;
83  
84          endSi = state.getEnd(tis[ofs]);
85          nextChar = ((Range) state.getInput(tis[ofs])).getLast() + 1;
86          for (i = ofs + 1; i < tis.length; i++) {
87              if (state.getEnd(tis[i]) != endSi) {
88                  break;
89              }
90              tmp = (Range) state.getInput(tis[i]);
91              if (tmp.getFirst() != nextChar) {
92                  break;
93              }
94              nextChar = tmp.getLast() + 1;
95          }
96          return i - ofs;
97      }
98  
99      private static Range firstToLast(State state, int[] tis, int ofs, int width) {
100         Range left, right;
101 
102         left = (Range) state.getInput(tis[ofs]);
103         right = (Range) state.getInput(tis[ofs + width - 1]);
104         return new Range(left.getFirst(), right.getLast());
105     }
