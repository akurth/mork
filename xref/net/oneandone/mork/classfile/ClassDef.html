40      /**
41       * Create a new class file with the specified name and
42       * super class name. Add interfaces, fields, methods and attributes
43       * my modifying the respective public fields.
44       */
45      public ClassDef(ClassRef thisClass, ClassRef superClass) {
46          this.accessFlags = new HashSet<Access>();
47          accessFlags.add(Access.PUBLIC);
48          accessFlags.add(Access.SUPER);
49          this.thisClass = thisClass;
50          this.superClass = superClass;
51          this.interfaces = new ArrayList<ClassRef>();
52          this.fields = new ArrayList<FieldDef>();
53          this.methods = new ArrayList<MethodDef>();
54          this.attributes = new ArrayList<Attribute>();
55      }

57      /**
58       * Create a class file from the specified input stream.
59       */
60      public ClassDef(Input src) throws IOException {
61          int i, max;
62  
63          interfaces = new ArrayList<ClassRef>();
64          fields = new ArrayList<FieldDef>();
65          methods = new ArrayList<MethodDef>();
66          attributes = new ArrayList<Attribute>();
67  
68          accessFlags = Access.fromFlags(src.readU2(), true);
69          thisClass = src.readClassRef();
70          try {
71              superClass = src.readClassRef();
72          } catch (NullPointerException e) {
73              if (!thisClass.isJavaLangObject()) {
74                  throw new RuntimeException(
75                      "missing base class for " + thisClass);
76              }
77              superClass = null;
78          }
79          max = src.readU2();
80          for (i = 0; i < max; i++) {
81              interfaces.add(src.readClassRef());
82          }
83          max = src.readU2();
84          for (i = 0; i < max; i++) {
85              fields.add(new FieldDef(src));
86          }
87          max = src.readU2();
88          for (i = 0; i < max; i++) {
89              methods.add(new MethodDef(src));
90          }
91          max = src.readU2();
92          for (i = 0; i < max; i++) {
93              attributes.add(Attribute.create(src));
94          }
95      }

105     public MethodDef lookupMethod(Set<Access> accessFlags, String name, ClassRef[] argumentTypes, ClassRef returnType,
106             List<ClassRef> exceptions) {
107         for (MethodDef m : methods) {
108             if (m.accessFlags.equals(accessFlags) && m.name.equals(name)
109                     && Arrays.equals(m.argumentTypes, argumentTypes)
110                     && m.returnType.equals(returnType)
111                     && m.getExceptions().equals(exceptions)) {
112                 return m;
113             }
114         }
115         return null;
116     }

151     /**
152      * Write this class file to the specified output stream.
153      */
154     public void write(Output dest) throws IOException {
155         int i, max;
156 
157         dest.writeU2(Access.toFlags(accessFlags));
158         dest.writeClassRef(thisClass);
159         if (superClass != null) {
160             dest.writeClassRef(superClass);
161         } else {
162             dest.writeU2(0);
163         }
164 
165         max = interfaces.size();
166         dest.writeU2(max);
167         for (i = 0; i < max; i++) {
168             dest.writeClassRef(interfaces.get(i));
169         }
170         max = fields.size();
171         dest.writeU2(max);
172         for (i = 0; i < max; i++) {
173             fields.get(i).write(dest);
174         }
175         max = methods.size();
176         dest.writeU2(max);
177         for (i = 0; i < max; i++) {
178             methods.get(i).write(dest);
179         }
180         max = attributes.size();
181         dest.writeU2(max);
182         for (i = 0; i < max; i++) {
183             attributes.get(i).write(dest);
184         }
185     }

229     @Override
230     public String toString() {
231         StringBuilder result;
232         int i, max;
233 
234         result = new StringBuilder();
235         result.append("thisClass: " + thisClass + "\n");
236         result.append("superClass: " + superClass + "\n");
237         max = interfaces.size();
238         for (i = 0; i < max; i++) {
239             result.append("implements: " + interfaces.get(i) + "\n");
240         }
241         max = fields.size();
242         for (i = 0; i < max; i++) {
243             result.append("field: " + fields.get(i) + "\n");
244         }
245         max = methods.size();
246         for (i = 0; i < max; i++) {
247             result.append("method: " + methods.get(i) + "\n");
248         }
249         return result.toString();
250     }

252     public String toSignatureString() {
253         StringBuilder result;
254         int i, max;
255 
256         result = new StringBuilder();
257         result.append(Access.toPrefix(accessFlags));
258         result.append(getName());
259         max = interfaces.size();
260         if (!superClass.isJavaLangObject()) {
261             result.append(" extends ");
262             result.append(superClass.toString());
263         }
264         if (max > 0) {
265             result.append(" implements");
266             for (i = 0; i < max; i++) {
267                 result.append(' ');
268                 result.append(interfaces.get(i));
269             }
270         }
271         return result.toString();
272     }
