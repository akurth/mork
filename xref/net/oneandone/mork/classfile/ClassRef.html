196     /**
197      * @return ClassRef nextOfs
198      */
199     public static Object[] forFieldDescriptor(String descriptor, int ofs, int length) {
200         int i, dimensions;
201         char c;
202         Type typeCode;
203         String name;
204         int nextOfs;
205 
206         dimensions = 0;
207         while ((ofs < length) && (descriptor.charAt(ofs) == '[')) {
208             dimensions++;
209             ofs++;
210         }
211         if (ofs >= length) {
212             throw new RuntimeException("illegal descriptor: " + descriptor);
213         }
214         c = descriptor.charAt(ofs);
215         if (c == Type.REFERENCE.descriptor) {
216             typeCode = Type.REFERENCE;
217             i = descriptor.indexOf(';', ofs);
218             if (i == -1) {
219                 throw new RuntimeException(descriptor + ofs);
220             }
221             name = descriptor.substring(ofs + 1, i).replace('/', '.');
222             nextOfs = i + 1;
223         } else {
224             typeCode = null;
225             for (i = 0; i < Type.PRIMITIVES.length; i++) {
226                 typeCode = Type.PRIMITIVES[i];
227                 if (typeCode.descriptor == c) {
228                     break;
229                 }
230             }
231             if (typeCode == null) {
232                 throw new RuntimeException("illegal descriptor char "
233                     + (int) c + ", ofs " + ofs + " in: " + descriptor);
234             }
235             name = typeCode.name;
236             nextOfs = ofs + 1;
237         }
238         return new Object[] { new ClassRef(name, dimensions, typeCode), nextOfs };
239     }

347     /**
348      * Gets the common base of two classes. Common base is the most special
349      * class both argument are assignable to. The common base for
350      * different primitive types is null; the common base for a primitive
351      * type and a reference type is null.
352      * @param   a  first class, may be null
353      * @param   b  second class
354      * @return  the common base; b, if a == null;
355      */
356     public static Class<?> commonBase(Class<?> a, Class<?> b) {
357         Class<?> result;
358         Class<?> ifc;
359 
360         if (b == null) {
361             throw new IllegalArgumentException();
362         } else if (a == null) {
363             return b;
364         } else {
365             result = commonSuperClass(a, b);
366             if (Object.class.equals(result)) {
367                 ifc = commonInterfaces(a, b);
368                 if (ifc != null) {
369                     result = ifc;
370                 }
371             }
372             return result;
373         }
374     }
375 
376     private static Class<?> commonInterfaces(Class<?> a, Class<?> b) {
377         for (Class<?> left : a.getInterfaces()) {
378             for (Class<?> right : b.getInterfaces()) {
379                 if (left.equals(right)) {
380                     return left;
381                 }
382             }
383         }
384         return null;
385     }
386 
387     private static Class<?> commonSuperClass(Class<?> a, Class<?> b) {
388         if (a.isAssignableFrom(b)) {
389             return a;
390         } else if (b.isAssignableFrom(a)) {
391             return b;
392         } else {
393             Class<?> c = b.getSuperclass();
394             if (c == null) {
395                 return null; // primitive type, Void, or Interface
396             } else {
397                 return commonSuperClass(a, c);
398             }
399         }
400     }

403      * Writes a class Object.
404      * @param  out target to write to
405      * @param  cl  class to be written
406      */
407     public static void write(ObjectOutput out, Class<?> cl) throws IOException {
408         int dim;
409 
410         if (cl == null) {
411             out.writeByte(-1);
412         } else {
413             dim = 0;
414             while (cl.isArray()) {
415                 dim++;
416                 cl = cl.getComponentType();
417             }
418 
419             if (dim > Byte.MAX_VALUE) {
420                 throw new RuntimeException("to many dimensions");
421             }
422 
423             out.writeByte((byte) dim);
424             out.writeUTF(cl.getName());
425         }
426     }

428     /**
429      * Reads a class Object.
430      * @param  in  source to read from
431      * @return the Class read
432      */
433     public static Class<?> read(ObjectInput in) throws java.io.IOException {
434         byte dim;
435         Class<?> cl;
436         String name;
437 
438         dim = in.readByte();
439         if (dim == -1) {
440             return null;
441         } else {
442             name = in.readUTF();
443             cl = classFind(name);
444             if (cl == null) {
445                 throw new RuntimeException("can't load class " + name);
446             }
447 
448             while (dim-- > 0) {
449                 cl = net.oneandone.sushi.util.Arrays.getArrayClass(cl);
450             }
451             return cl;
452         }
453     }

460     public static void writeClasses(ObjectOutput out, Class<?>[] types)
461             throws java.io.IOException {
462         int i;
463 
464         if (types.length > Byte.MAX_VALUE) {
465             throw new RuntimeException("to many dimensions");
466         }
467         out.writeByte((byte) types.length);
468         for (i = 0; i < types.length; i++) {
469             write(out, types[i]);
470         }

555     public boolean isArrayDefaultElement(Object obj) {
556         Type typeCode;
557 
558         typeCode = getTypeCode();
559         switch (typeCode.id) {
560         case T_BOOLEAN:
561             return !((Boolean) obj).booleanValue();
562         case T_BYTE:
563         case T_SHORT:
564         case T_INT:
565         case T_LONG:
566             return ((Number) obj).longValue() == 0;
567         case T_CHAR:
568             return ((Character) obj).charValue() == 0;
569         case T_DOUBLE:
570         case T_FLOAT:
571             return ((Number) obj).doubleValue() == 0.0;
572         case T_REFERENCE:
573             return obj == null;
574         default:
575             throw new RuntimeException("not supported: " + typeCode.id);
576         }
