62      public Code(Input src) throws IOException {
63          this();
64  
65          Instruction instr;
66          int i;
67          int max;
68          int loadedStack;
69          int computedStack;
70  
71          src.readU4(); // attribute length is not used
72  
73          loadedStack = src.readU2();
74          locals = src.readU2();
75          max = src.readU4();
76          src.openCode(this);
77          while (src.getOfs() < max) {
78              instr = Instruction.read(src);
79              instructions.add(instr);
80          }
81          codeSize = src.getOfs();
82          max = instructions.size();
83          for (i = 0; i < max; i++) {
84              instr = instructions.get(i);
85              instr.ofsToIdx(this);
86          }
87          max = src.readU2();
88          for (i = 0; i < max; i++) {
89              exceptions.add(new ExceptionInfo(src));
90          }
91          max = src.readU2();
92          for (i = 0; i < max; i++) {
93              attributes.add(Attribute.create(src));
94          }
95          src.closeCode();
96  
97          computedStack = calcStackSize();
98          if (loadedStack < computedStack) {
99              LOGGER.warning(
100                 "loaded stack size differs from computed: " +
101                 "loaded: " + loadedStack + " < computed: " + computedStack);
102         }
103     }

122     @Override
123     public void write(Output dest) throws IOException {
124         int i, max;
125         Instruction instr;
126         ExceptionInfo info;
127         Attribute attr;
128         int start;
129 
130         layout(dest);
131         if (codeSize > 0xffff) {
132             throw new IllegalStateException("code segment exceeds 64k");
133         }
134         dest.writeUtf8(name);
135         start = dest.writeSpace(4);
136         dest.writeU2(calcStackSize());
137         dest.writeU2(locals);
138         dest.writeU4(codeSize);
139         dest.openCode(this);
140         max = instructions.size();
141         for (i = 0; i < max; i++) {
142             instr = (Instruction) instructions.get(i);
143             instr.write(dest);
144         }
145         max = exceptions.size();
146         dest.writeU2(max);
147         for (i = 0; i < max; i++) {
148             info = (ExceptionInfo) exceptions.get(i);
149             info.write(dest);
150         }
151         max = attributes.size();
152         dest.writeU2(max);
153         for (i = 0; i < max; i++) {
154             attr = (Attribute) attributes.get(i);
155             attr.write(dest);
156         }
157         dest.closeCode();
158         dest.writeFixup(start, dest.getGlobalOfs() - (start + 4));
159     }

161     /** compute ofs for all instructions. */
162     private void layout(Output dest) {
163         Instruction instr;
164         int instrSize, varSize;
165         IntArrayList vars;  // indexes of instructions with variable length
166         IntArrayList lens;  // lengths of var-len instructions
167         int i, j, k, ofs;
168         int shrink, len;
169         boolean changes;
170 
171         instrSize = instructions.size();
172         vars = new IntArrayList();
173         lens = new IntArrayList();
174         ofs = 0;
175         for (i = 0; i < instrSize; i++) {
176             instr = instructions.get(i);
177             instr.ofs = ofs;
178             len = instr.getMaxLength(dest);
179             ofs += len;
180             if (instr.type.isVariable()) {
181                 vars.add(i);
182                 lens.add(len);
183             }
184         }
185         codeSize = ofs;
186         varSize = vars.size();
187         do {
188             changes = false;
189             shrink = 0;
190             for (i = 0; i < varSize; i++) {
191                 j = vars.get(i);
192                 instr = instructions.get(j);
193                 len = instr.getVariableLength(this);
194                 if (len < lens.get(i)) {
195                     changes = true;
196                     shrink += (lens.get(i) - len);
197                     lens.set(i, len);
198                 }
199                 if (shrink > 0) {
200                     // relocate up to end or next var-instr
201                     for (k = (i + 1 == varSize)? instrSize - 1 : vars.get(i + 1);  k > j; k--) {
202                         instructions.get(k).ofs -= shrink;
203                     }
204                 }
205             }
206             codeSize -= shrink;
207         } while (changes);
208     }

224     public int findIdx(int ofs) {
225         // binary search
226         int low, high;
227         int idx;
228         int tmp;
229 
230         low = 0;
231         high = instructions.size() - 1;
232 
233         while (low <= high) {
234             idx = (low + high) / 2;
235             tmp = instructions.get(idx).ofs;
236             if (tmp < ofs) {
237                 low = idx + 1;
238             } else if (tmp > ofs) {
239                 high = idx - 1;
240             } else {
241                 return idx;
242             }
243         }
244         throw new RuntimeException("no such ofs: " + ofs);
245     }

267     public int resolveLabel(int idx) {
268         int trueIdx;
269 
270         if (idx < 0) {
271             trueIdx = labels.get(-idx);
272             if (trueIdx < 0) {
273                 throw new RuntimeException("undefined label: " + idx);
274             }
275             return trueIdx;
276         } else {
277             return idx;
278         }
279     }

301         StringBuilder result;
302         int i;

304 
305         result = new StringBuilder();
306         result.append("Code attribute: \n");

315             result.append('\t');
316             result.append(exceptions.get(i).toString());
317             result.append('\n');
318         }
319         for (i = 0; i < attributes.size(); i++) {
320             result.append('\t');
321             result.append(attributes.get(i).toString());
322             result.append('\n');
323         }
324         return result.toString();

430     // computed the stack size
431     private int calcStackSize() {
432         int i, max;
433         int result;
434         int[] startStack;
435         ExceptionInfo e;
436         int tmp;
437         int unreachable;
438         IntBitSet todo;
439         List<Jsr> jsrs;
440 
441         jsrs = Jsr.findJsrs(this);
442         startStack = new int[instructions.size()];
443         for (i = 0; i < startStack.length; i++) {
444             startStack[i] = -1;
445         }
446         startStack[0] = 0;
447         todo = new IntBitSet();
448         todo.add(0);
449         max = exceptions.size();
450         for (i = 0; i < max; i++) {
451             e = exceptions.get(i);
452             startStack[e.handler] = 1;
453             todo.add(e.handler);
454         }
455 
456         fillStack(jsrs, todo, startStack);
457 
458         result = 0;
459         unreachable = -1;
460         for (i = 0; i < startStack.length; i++) {
461             tmp = startStack[i];
462             if (tmp > result) {
463                 result = tmp;
464             }
465             if ((unreachable == -1) && (tmp == -1)) {
466                 unreachable = i;
467             }
468         }
469         if (unreachable != -1) {
470             // There are several class file in Sun JDK 1.3 with dead
471             // code, and they pass class file validation:
472             // (e.g. com.sun.corba.se.intneral.io.util.Arrays)
473 
474             LOGGER.warning("unreachable code, starting " + unreachable + "\n");
475         }
476         return result;
477     }

479     private void fillStack(List<Jsr> jsrs, IntBitSet todo, int[] startStack) {
480         int succSize;
481         Instruction instr;
482         int idx;
483         int i, max;
484         int tmp, succIdx;
485         IntArrayList succBuffer;
486 
487         while (true) {
488             idx = todo.first();
489             if (idx == -1) {
490                 return;
491             }
492             todo.remove(idx);
493 
494             instr = instructions.get(idx);
495             succSize = startStack[idx] + instr.getStackDiff();
496             if (succSize < 0) {
497                 throw new RuntimeException("stack size <0, idx: " + idx);
498             }
499             succBuffer = new IntArrayList();
500             instr.getSuccessors(jsrs, idx, this, succBuffer);
501             max = succBuffer.size();
502             for (i = 0; i < max; i++) {
503                 succIdx = succBuffer.get(i);
504                 tmp = startStack[succIdx];
505                 if (tmp == succSize) {
506                     // do nothing, end of recursion
507                 } else if (tmp == -1) {
508                     startStack[succIdx] = succSize;
509                     todo.add(succIdx);
510                 } else {
511                     throw new RuntimeException("wrong stack idx = " + succIdx
512                                                + ": " + tmp + "!=" + succSize + "\n" + toString());
513                 }
514             }
515         }
516     }
