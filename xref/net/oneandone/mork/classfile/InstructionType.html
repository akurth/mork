57      public void checkArgs(Object[] argValues) {
58          // TODO: add more tests
59  
60          switch (encoding) {
61          case SIMPLE:
62              checkSimpleArgs(argValues);
63              break;
64          case RT: // RETURN
65          case LV:
66              if (argValues.length != 1) {
67                  throw new RuntimeException("1 arg expected: " + this + " " +
68                                             argValues);
69              }
70              if (!(argValues[0] instanceof Integer)) {
71                  throw new RuntimeException("Integer arg expected: " +
72                                             this + " " + argValues[0]);
73              }
74              break;
75          case INC:
76              if (argValues.length != 2) {
77                  throw new RuntimeException("2 args expected: " + this + " " +
78                                             argValues);
79              }
80              if (!(argValues[0] instanceof Integer)) {
81                  throw new RuntimeException("Integer arg expected: " +
82                                             this + " " + argValues[0]);
83              }
84              if (!(argValues[1] instanceof Integer)) {
85                  throw new RuntimeException("Integer arg 2 expected: " +
86                                             this + " " + argValues[1]);
87              }
88              break;
89          case BRANCH:
90          case VBRANCH:
91              if (argValues.length != 1) {
92                  throw new RuntimeException("<>BRANCH: 1 arg expected: " +
93                                             argValues.length);
94              }
95              if (!(argValues[0] instanceof Integer)) {
96                  throw new RuntimeException("Integer arg expected: " +
97                                             this + " " + argValues[0]);
98              }
99              break;
100         case LS:
101             if (argValues.length != 3) {
102                 throw new RuntimeException("LS: 3 arg expected: " +
103                                            argValues.length);
104             }
105             if (!(argValues[0] instanceof Integer)) {
106                 throw new RuntimeException("Integer arg expected: " +
107                                            this + " " + argValues[0]);
108             }
109             if (!(argValues[1] instanceof IntArrayList)) {
110                 throw new RuntimeException("IntArrayList arg expected: " +
111                                            this + " " + argValues[1]);
112             }
113             if (!(argValues[2] instanceof IntArrayList)) {
114                 throw new RuntimeException("IntArrayList arg expected: " +
115                                            this + " " + argValues[2]);
116             }
117             break;
118         case TS:
119             if (argValues.length != 4) {
120                 throw new RuntimeException("TS: 4 arg expected: " +
121                                            argValues.length);
122             }
123             if (!(argValues[0] instanceof Integer)) {
124                 throw new RuntimeException("Integer arg expected: " +
125                                            this + " " + argValues[0]);
126             }
127             if (!(argValues[1] instanceof Integer)) {
128                 throw new RuntimeException("Integer arg expected: " +
129                                            this + " " + argValues[1]);
130             }
131             if (!(argValues[2] instanceof Integer)) {
132                 throw new RuntimeException("Integer arg expected: " +
133                                            this + " " + argValues[2]);
134             }
135             if (!(argValues[3] instanceof IntArrayList)) {
136                 throw new RuntimeException("IntArrayList arg expected: " +
137                                            this + " " + argValues[3]);
138             }
139             break;
140         case CNST:
141             if (argValues.length != 1) {
142                 throw new RuntimeException("LDC: 1 arg expected: " +
143                                            argValues.length);
144             }
145             break;
146         default:
147             throw new RuntimeException("illegal encoding: " + encoding);
148         }

151     public void checkSimpleArgs(Object[] argValues) {
152         int i;
153         Object v;
154 
155         if (args.length != argValues.length) {
156             throw new RuntimeException("illegal argument count: "
157                                        + this + ":" + argValues.length);
158         }
159         for (i = 0; i < args.length; i++) {
160             v = argValues[i];
161             switch (args[i]) {
162             case REFTYPEREF:
163                 if (!(v instanceof ClassRef)) {
164                     throw new RuntimeException("ClassRef expected: " + this
165                                                + v);
166                 }
167                 break;
168             case FIELDREF:
169                 if (!(v instanceof FieldRef)) {
170                     throw new RuntimeException("FieldRef expected: " + this
171                                                + v);
172                 }
173                 break;
174             case IFMETHOD:
175                 if (!(v instanceof MethodRef)) {
176                     throw new RuntimeException("MethodRef expected: " + this
177                                                + v);
178                 }
179                 break;
180             case METHODREF:
181                 if (!(v instanceof MethodRef)) {
182                     throw new RuntimeException("MethodRef expected: " + this
183                                                + v);
184                 }
185                 break;
186             case BYTE:
187             case TYPE_BYTE:
188                 if (!(v instanceof Integer)) {
189                     throw new RuntimeException("Integer expected: " + this
190                                                + v);
191                 }
192                 break;
193             default:
194                 throw new RuntimeException("illegal argument type: " +
195                                            args[i]);
196             }
197         }
198     }

200     private void calcLength() {
201         int i;
202 
203         if (encoding != SIMPLE) {
204             length = -1;
205         } else {
206             length = 1;
207             for (i = 0; i < args.length; i++) {
208                 switch (args[i]) {
209                 case BYTE:
210                 case TYPE_BYTE:
211                     length += 1;
212                     break;
213                 case REFTYPEREF:
214                 case FIELDREF:
215                 case METHODREF:
216                     length += 2;
217                     break;
218                 case IFMETHOD:
219                     // because of "<size> 0" following the index
220                     length += 4;
221                     break;
222                 default:
223                     throw new IllegalArgumentException(
224                        name + ": illegal arg type: " + args[i]);
225                 }
226             }
227         }
228     }

238     public void ofsToIdx(Code context, int ofs, Object[] argValues) {
239         int i;
240         IntArrayList branchesW;
241 
242         switch (encoding) {
243         case TS:
244             i = ((Integer) argValues[0]).intValue();
245             argValues[0] = new Integer(context.findIdx(ofs + i));
246             branchesW = (IntArrayList) argValues[3];
247             for (i = 0; i < branchesW.size(); i++) {
248                 branchesW.set(i, context.findIdx(ofs + branchesW.get(i)));
249             }
250             break;
251         case LS:
252             i = ((Integer) argValues[0]).intValue();
253             argValues[0] = new Integer(context.findIdx(ofs + i));
254             branchesW = (IntArrayList) argValues[2];
255             for (i = 0; i < branchesW.size(); i++) {
256                 branchesW.set(i, context.findIdx(ofs + branchesW.get(i)));
257             }
258             break;
259         case BRANCH:
260         case VBRANCH:
261             i = ((Integer) argValues[0]).intValue();
262             argValues[0] = new Integer(context.findIdx(ofs + i));
263             break;
264         default:
265             // do nothing
266             break;
267         }
268     }

276     public void write(Output dest, Object[] values) throws IOException {
277         int i, i2;
278         IntArrayList ofss;
279         IntArrayList keys;
280         int ofs;
281 
282         switch (encoding) {
283         case SIMPLE:
284             writeSimple(dest, values);
285             break;
286         case CNST:
287             writeCNST(dest, values[0]);
288             break;
289         case BRANCH:
290             i = branch(dest.getOfs(), ((Integer) values[0]).intValue(), dest);
291             if ((Short.MIN_VALUE <= i) && (i <= Short.MAX_VALUE)) {
292                 dest.writeU1(opcode);
293                 dest.writeS2(i);
294             } else {
295                 dest.writeU1(args[0]);
296                 dest.writeS2(8);
297                 dest.writeU1(Bytecodes.GOTO_W);
298                 dest.writeU4(i - 3);
299             }
300             break;
301         case VBRANCH:
302             i = branch(dest.getOfs(), ((Integer) values[0]).intValue(), dest);
303             if ((Short.MIN_VALUE <= i) && (i <= Short.MAX_VALUE)) {
304                 dest.writeU1(opcode);
305                 dest.writeS2(i);
306             } else {
307                 dest.writeU1(args[0]);
308                 dest.writeU4(i);
309             }
310             break;
311         case TS:
312             ofs = dest.getOfs();
313             dest.writeU1(opcode);
314             dest.writePad();
315             dest.writeU4(branch(ofs, ((Integer) values[0]).intValue(), dest));
316             dest.writeU4(((Integer) values[1]).intValue());
317             dest.writeU4(((Integer) values[2]).intValue());
318             ofss = (IntArrayList) values[3];
319             for (i = 0; i < ofss.size(); i++) {
320                 dest.writeU4(branch(ofs, ofss.get(i), dest));
321             }
322             break;
323         case LS:
324             ofs = dest.getOfs();
325             dest.writeU1(opcode);
326             dest.writePad();
327             dest.writeU4(branch(ofs, ((Integer) values[0]).intValue(), dest));
328             keys = (IntArrayList) values[1];
329             ofss = (IntArrayList) values[2];
330             dest.writeU4(keys.size());
331             for (i = 0; i < keys.size(); i++) {
332                 dest.writeU4(keys.get(i));
333                 dest.writeU4(branch(ofs, ofss.get(i), dest));
334             }
335             break;
336         case LV:
337             i = ((Integer) values[0]).intValue();
338             if (i < 0) {
339                 throw new RuntimeException("LV < 0 " + i);
340             } else if (i <= 3) {
341                 dest.writeU1(args[i]);
342             } else if (i <= 255) {
343                 dest.writeU1(opcode);
344                 dest.writeU1(i);
345             } else if (i <= Character.MAX_VALUE) {
346                 dest.writeU1(WIDE);
347                 dest.writeU1(opcode);
348                 dest.writeU2(i);
349             } else {
350                 throw new RuntimeException();
351             }
352             break;
353         case RT:
354             i = ((Integer) values[0]).intValue();
355             if (i <= 255) {
356                 dest.writeU1(opcode);
357                 dest.writeU1(i);
358             } else {
359                 dest.writeU1(WIDE);
360                 dest.writeU1(opcode);
361                 dest.writeU2(i);
362             }
363             break;
364         case INC:
365             i = ((Integer) values[0]).intValue();
366             i2 = ((Integer) values[1]).intValue();
367             if ((i <= 255) &&
368                 (Byte.MIN_VALUE <= i2) && (i2 <= Byte.MAX_VALUE)) {
369                 dest.writeU1(opcode);
370                 dest.writeU1(i);
371                 dest.writeS1(i2);
372             } else {
373                 dest.writeU1(WIDE);
374                 dest.writeU1(opcode);
375                 dest.writeU2(i);
376                 dest.writeS2(i2);
377             }
378             break;
379         default:
380             throw new RuntimeException("unknown encoding: " + encoding);
381         }
382     }

385     private void writeSimple(Output dest, Object[] values) throws IOException {
386         int i;
387         Object val;
388         MethodRef ifc;
389 
390         dest.writeU1(opcode);
391         for (i = 0; i < args.length; i++) {
392             val = values[i];
393             switch (args[i]) {
394             case REFTYPEREF:
395                 dest.writeClassRef((ClassRef) val);
396                 break;
397             case FIELDREF:
398                 dest.writeFieldRef((FieldRef) val);
399                 break;
400             case METHODREF:
401                 dest.writeClassMethodRef((MethodRef) val);
402                 break;
403             case IFMETHOD:
404                 ifc = (MethodRef) val;
405                 dest.writeInterfaceMethodRef((MethodRef) val);
406                 dest.writeU1(1 + ifc.argSize()); // 1 for "this"
407                 dest.writeU1(0);
408                 break;
409             case BYTE:
410             case TYPE_BYTE:
411                 dest.writeU1(((Integer) val).intValue());
412                 break;
413             default:
414                 throw new RuntimeException("illegal argType: " + args[i]);
415             }
416         }
417     }

419     public void writeCNST(Output dest, Object val) throws IOException {
420         int e, ae;
421 
422         e = findConstEncoding(dest, val);
423         dest.writeU1(e);
424         ae = Set.ENCODING[e].args[0];
425         switch (ae) {
426         case AE_U1:
427             dest.writeU1(((Integer) val).intValue());
428             break;
429         case AE_S1:
430             dest.writeS1(((Integer) val).intValue());
431             break;
432         case AE_U2:
433             dest.writeU2(((Integer) val).intValue());
434             break;
435         case AE_S2:
436             dest.writeS2(((Integer) val).intValue());
437             break;
438         case AE_CNST:
439             if (val instanceof Integer) {
440                 dest.writeShortInt(((Integer) val).intValue());
441             } else if (val instanceof Float) {
442                 dest.writeShortFloat(((Float) val).floatValue());
443             } else {
444                 dest.writeShortString((String) val);
445             }
446             break;
447         case AE_CNST_W:
448             if (val instanceof Integer) {
449                 dest.writeInt(((Integer) val).intValue());
450             } else if (val instanceof Float) {
451                 dest.writeFloat(((Float) val).floatValue());
452             } else {
453                 dest.writeString((String) val);
454             }
455             break;
456         case AE_CNST2_W:
457             if (val instanceof Long) {
458                 dest.writeLong(((Long) val).longValue());
459             } else {
460                 dest.writeDouble(((Double) val).doubleValue());
461             }
462             break;
463         default:
464             if (ae <= AE_I_LAST) {
465                 // no explicit arguments
466             } else {
467                 throw new IllegalArgumentException(
468                     name + ": illegal arg encoding: " + ae);
469             }
470         }
471     }

487     public int getVariableLength(Code context, int ofs, Object[] argValues) {
488         int tmp;
489 
490         switch (encoding) {
491         case TS:
492             return 1 + IO.padSize(ofs + 1)
493                 + 3*4 + ((IntArrayList) argValues[3]).size() * 4;
494         case LS:
495             return 1 + IO.padSize(ofs + 1)
496                 + 2*4 + ((IntArrayList) argValues[1]).size() * 4 * 2;
497         case BRANCH:
498             tmp = context.findOfs(((Integer) argValues[0]).intValue()) - ofs;
499             if ((Short.MIN_VALUE <= tmp) && (tmp <= Short.MAX_VALUE)) {
500                 return 3;
501             } else {
502                 return 8;
503             }
504         case VBRANCH:
505             tmp = context.findOfs(((Integer) argValues[0]).intValue()) - ofs;
506             if ((Short.MIN_VALUE <= tmp) && (tmp <= Short.MAX_VALUE)) {
507                 return 3;
508             } else {
509                 return 5;
510             }
511         default:
512             throw new RuntimeException("not of variable length: " + encoding);
513         }

517         int i, i2;
518 
519         switch (encoding) {

521             return length;
522         case CNST:
523             return constLength(dest, argValues[0]);
524         case BRANCH:

526         case VBRANCH:

529         case TS:

531             return 1 + 3 + 4*3 + ((IntArrayList) argValues[3]).size() * 4;
532         case LS:
533             // maximum pad size: 3
534             return 1 + 3 + 4*2 + ((IntArrayList) argValues[1]).size() * 2 * 4;
535         case LV:
536             i = ((Integer) argValues[0]).intValue();
537             if (i < 0) {
538                 throw new RuntimeException("LV to small: " + i);
539             } else if (i <= 3) {
540                 return 1;
541             } else if (i <= 255) {
542                 return 2;
543             } else if (i <= Character.MAX_VALUE) {
544                 return 4;
545             } else {
546                 throw new RuntimeException("LV to big: " + i);
547             }
548         case RT:
549             i = ((Integer) argValues[0]).intValue();
550             if (i <= 255) {
551                 return 2;
552             } else {
553                 // wide
554                 return 4;
555             }
556         case INC:
557             i = ((Integer) argValues[0]).intValue();
558             i2 = ((Integer) argValues[1]).intValue();
559             if ((i <= 255) &&

602     private int findConstEncoding(Output dest, Object cnst) {
603         boolean twice;
604         int idx;
605         int v;
606 
607         if (cnst == null) {
608             return ACONST_NULL;
609         } else if (cnst instanceof Integer) {
610             v = ((Integer) cnst).intValue();
611             switch (v) {
612             case -1:
613                 return ICONST_M1;
614             case 0:
615                 return ICONST_0;
616             case 1:
617                 return ICONST_1;
618             case 2:
619                 return ICONST_2;
620             case 3:
621                 return ICONST_3;
622             case 4:
623                 return ICONST_4;
624             case 5:
625                 return ICONST_5;
626             default:
627                 if ((Byte.MIN_VALUE <= v) && (v <= Byte.MAX_VALUE)) {
628                     return BIPUSH;
629                 } else if ((Short.MIN_VALUE <= v) && (v <= Short.MAX_VALUE)) {
630                     return SIPUSH;
631                 } else {
632                     twice = false;
633                     idx = dest.addIfNew(CONSTANT_INTEGER, cnst);
634                     // don't return
635                 }
636             }
637         } else if (cnst instanceof String) {
638             twice = false;
639             idx = dest.addIfNew(CONSTANT_STRING, cnst);
640             // don't return
641         } else if (cnst instanceof Long) {
642             if (IMPLICIT[AE_I_L0].equals(cnst)) {
643                 return LCONST_0;
644             } else if (IMPLICIT[AE_I_L1].equals(cnst)) {
645                 return LCONST_1;
646             } else {
647                 twice = true;
648                 idx = dest.addIfNew(CONSTANT_LONG, cnst);
649                 // don't return
650             }
651         } else if (cnst instanceof Float) {
652             if (IMPLICIT[AE_I_F0].equals(cnst)) {
653                 return FCONST_0;
654             } else if (IMPLICIT[AE_I_F1].equals(cnst)) {
655                 return FCONST_1;
656             } else {
657                 twice = false;
658                 idx = dest.addIfNew(CONSTANT_FLOAT, cnst);
659             }
660         } else if (cnst instanceof Double) {
661             if (IMPLICIT[AE_I_D0].equals(cnst)) {
662                 return DCONST_0;
663             } else if (IMPLICIT[AE_I_D1].equals(cnst)) {
664                 return DCONST_1;
665             } else {
666                 twice = true;
667                 idx = dest.addIfNew(CONSTANT_DOUBLE, cnst);
668                 // don't return;
669             }
670         } else {
671             throw new RuntimeException("illegal constant: " + cnst);
672         }
673 
674         if (!twice && (idx <= 255)) {
675             return LDC;
676         } else {
677             return (twice)? LDC2_W : LDC_W;
678         }
679     }
