42      /**
43       * @param mapperName        mapper name as specified in the map file.
44       * @param src               file that specified the mapper
45       * @param explicitOutputDir "-d", points to a directory or null.
46       */
47      public void run(Mapper mapper, String mapperName, File src, File explicitOutputDir) throws IOException {
48          String baseName;
49          File outputDir;  // directory to write all class files to
50  
51          String mapperClassName;
52          File mapperFile;
53          String functionClassName;
54          String functionFileBase;
55  
56          ClassDef c;
57          FunctionCompiler fc;
58  
59          baseName = mapperName.substring(mapperName.lastIndexOf('.') + 1);
60            // also ok of idx -1
61          outputDir = outputDir(src, explicitOutputDir, mapperName);
62  
63          mapperClassName = mapperName;
64          mapperFile = new File(outputDir, baseName + ".class");
65          functionClassName = mapperName + "Functions";
66          functionFileBase = new File(outputDir, baseName + "Functions").getPath();
67  
68          fc = new FunctionCompiler(functionClassName);
69          customs[0] = fc;
70          output.verbose("translating " + mapperClassName);
71          c = translate(mapper, mapperClassName);
72  
73          output.verbose("writing " + mapperFile);
74          try {
75              Output.save(c, mapperFile);
76          } catch (IOException e) {
77              output.error(mapperFile.toString(), "write failed: " + e);
78          }
79          try {
80              fc.save(functionFileBase);
81          } catch (IOException e) {
82              output.error(functionFileBase.toString(), "write failed: " + e);
83          }
84          output.verbose("done");
85      }

87      /** creates new directory if necessary. */
88      public File outputDir(File src, File explicitOutputDir, String mapperName) throws IOException {
89          File outputDir;
90          int prev;
91          int idx;
92          File subDir;
93  
94          if (explicitOutputDir == null) {
95              outputDir = src.getParentFile();
96          } else {
97              outputDir = explicitOutputDir;
98              idx = mapperName.indexOf('.');
99              prev = 0;
100             while (idx != -1) {
101                 subDir = new File(outputDir, mapperName.substring(prev, idx));
102                 if (!subDir.isDirectory()) {
103                     if (!subDir.mkdir()) {
104                         throw new IOException("cannot create directory: " + subDir);
105                     }
106                 }
107                 prev = idx + 1;
108                 idx = mapperName.indexOf('.', prev);
109                 outputDir = subDir;
110             }
111         }
112         return outputDir;
113     }

115     private ClassDef translate(Mapper mapper, String className) {
116         ClassDef result;
117         ObjectCompiler compiler;
118         Code code;
119 
120         code = new Code();
121         code.locals = 1; // this
122         result = createClass(className, code);
123         compiler = new ObjectCompiler(code, code.allocate(ClassRef.INT), customs, result);
124         code.emit(LDC, 2);
125         code.emit(ANEWARRAY, ClassRef.OBJECT); // this, the new Mapper
126         code.emit(DUP);
127         code.emit(LDC, 0);
128         compiler.run(mapper.getParser());
129         code.emit(AASTORE);
130         code.emit(DUP);
131         code.emit(LDC, 1);
132         compiler.run(mapper.getSemantics());
133         code.emit(AASTORE);
134         code.emit(ARETURN);
135         return result;
136     }

138     private ClassDef createClass(String className, Code code) {
139         ClassDef result;
140 
141         result = new ClassDef(new ClassRef(className), ClassRef.OBJECT);
142         result.addMethod(Access.fromArray(Access.PUBLIC, Access.STATIC),
143                          ClassRef.OBJECT, "load", ClassRef.NONE, code);
144         return result;
145     }
