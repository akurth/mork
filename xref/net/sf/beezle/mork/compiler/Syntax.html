36  /**
37   * Grammar syntax specification. Represents a grammar syntax file with
38   * parser and scanner section.
39   *
40   * Design issues: whitespace handling is performed in the parser, not in the
41   * scanner. This is because I might a features access the "[text]" of white space.
42   */
43  public class Syntax {
44      public static final String CONFLICT = "lr(k) conflict:\n";
45  
46      private Grammar grammar;
47      private boolean priorities;
48      private IntBitSet whiteSymbols;
49      private Rule[] scannerRules;
50  
51      public Syntax(StringArrayList symbolTable, Rule[] parserRules, boolean priorities, IntBitSet whiteSymbols, Rule[] scannerRules) throws GenericException {
52          if (parserRules.length == 0) {
53              throw new IllegalArgumentException();
54          }
55          this.grammar = Ebnf.translate(parserRules, symbolTable);
56          this.priorities = priorities;
57          if (whiteSymbols != null) {
58              this.whiteSymbols = whiteSymbols;
59          } else {
60              this.whiteSymbols = new IntBitSet();
61          }
62          this.scannerRules = scannerRules;
63      }
64  
65      public Grammar getGrammar() {
66          return grammar;
67      }
68  
69      /**
70       * Translate specification.
71       *
72       * @return null for errors.
73       */
74      public Parser translate(int k, int threadCount, Output output) throws GenericException {
75          FABuilder builder;
76          long started;
77          PDA pda;
78          ParserTable parserTable;
79          Map<Integer, PrefixSet> firsts;
80          ScannerFactory scannerFactory;
81          IntBitSet usedTerminals;
82          IntBitSet usedSymbols;
83          int symbolCount;
84          StringArrayList symbolTable;
85          ConflictHandler handler;
86          ConflictResolver[] resolvers;
87  
88          started = System.currentTimeMillis();
89          output.verbose("computing firsts");
90          firsts = grammar.firsts(k);
91          output.verbose("creating pda, " + threadCount + " threads");
92          pda = PDA.create(grammar, firsts, k, threadCount);
93          output.verbose("done: " + pda.size() + " states, " + (System.currentTimeMillis() - started) + " ms");
94          symbolCount = Math.max(grammar.getSymbolCount(), whiteSymbols.last() + 1);
95          handler = new ConflictHandler(grammar);
96          parserTable = pda.createTable(symbolCount, handler);
97          parserTable.addWhitespace(whiteSymbols);
98          symbolTable = grammar.getSymbolTable();
99          if (output.listing != null) {
100             output.listing.println("\nSymbols:");
101             output.listing.println(symbolTable.toString());
102             output.listing.println("\nGrammar:");
103             output.listing.println(grammar.toString());
104             output.listing.println("\nFirst Sets:");
105             for (Map.Entry<Integer, PrefixSet> entry : firsts.entrySet()) {
106                 output.listing.println(symbolTable.get(entry.getKey()) + ":\t" + entry.getValue().toString(symbolTable));
107             }
108             output.listing.println("\nAutomaton:");
109             pda.print(output.listing);
110         }
111         if (output.statistics != null) {
112             output.statistics.println();
113             output.statistics.println("parser statistics");
114             output.statistics.println("  states: " + pda.size());
115             output.statistics.println("  table: [symbols=" + parserTable.getSymbolCount() + "][states=" + parserTable.getStateCount() + "]");
116             output.statistics.println("  lr(1) conflicts: " + handler.resolvers());
117             pda.statistics(output.statistics);
118         }
119 
120         resolvers = handler.report(output, grammar);
121 
122         // free memory before computing FA
123         pda = null;
124 
125         output.verbose("processing scanner section");
126 
127         usedTerminals = new IntBitSet();
128         grammar.getUsedTerminals(usedTerminals);
129         usedTerminals.addAll(whiteSymbols);
130 
131         output.verbose("generating scanner");
132         builder = FABuilder.run(scannerRules, usedTerminals, symbolTable, output.verbose);
133         output.listing("inline symbols: " + builder.getInlines());
134 
135         if (priorities) {
136             output.verbose("use priorities");
137             Modes.resolveScannerConflicts(builder.getFA(), scannerRules);
138         }
139         scannerFactory = ScannerFactory.create(
140             builder.getFA(), builder.getErrorState(), parserTable, whiteSymbols, output.verbose, output.listing);
141 
142         if (output.statistics != null) {
143             output.statistics.println();
144             output.statistics.println("scanner statistics");
145             output.statistics.println("  fa states : " + builder.getFA().size());
146             output.statistics.println("  table: char[" + scannerFactory.size() + "]");
147         }
148         output.verbose("scanner done");
149 
150         usedSymbols = new IntBitSet(whiteSymbols);
151         usedSymbols.addAll(builder.getInlines());
152         grammar.check(grammar.getStart(), usedSymbols, symbolTable.toList());
153 
154         return new Parser(parserTable, resolvers, scannerFactory);
155     }
