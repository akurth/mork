96      public static IntBitSet symbolSet(int[] lst) throws ActionException {
97          IntBitSet result;
98          int i, sym;
99  
100         result = new IntBitSet();
101         for (i = 0; i < lst.length; i++) {
102             sym = lst[i];
103             if (result.contains(sym)) {
104                 throw new ActionException("duplicate symbol: " + sym);
105             }
106             result.add(sym);
107         }
108         return result;
109     }

116     public static Definition createDefinition(
117         Grammar grm, StringArrayList symbolTable, int symbol, String name, Object constr)
118         throws GenericException {
119         boolean main;
120         IntBitSet terminals;
121 
122         main = (symbolTable.indexOf(name) == symbol);
123         if (grm.isTerminal(symbol)) {
124             terminals = new IntBitSet();
125             grm.getUsedTerminals(terminals);
126             // TODO: this computation is redundant to the computation of inline symbols in
127             // GrammarSyntax. However, inline symbols are not available before the end of parsing ...
128             if (!terminals.contains(symbol)) {
129                 throw new GenericException(ILLEGAL_INLINE);
130             }
131         }
132         return new Definition(main, grm, symbol, name, constr);
133     }

182     public static Object normalPath(Definition source, Specification spec, Object[] steps) throws GenericException {
183         int[] moves;
184         int[] symbols;
185         int i;
186         String str;
187         Definition target;
188         String[] symbolsOrAttributes;
189         StringArrayList symbolTable;
190 
191         symbolTable = spec.getSyntax().getGrammar().getSymbolTable();
192         moves = getMoves(steps);
193         symbolsOrAttributes = getSymbolsOrAttributes(steps);
194         symbols = new int[symbolsOrAttributes.length - 1];
195         for (i = 0; i < symbols.length; i++) {
196             str = symbolsOrAttributes[i];
197             target = spec.lookup(str);
198             if (target != null) {
199                 symbols[i] = target.getAttribute().symbol;
200             } else {
201                 symbols[i] = symbolTable.indexOf(str);
202                 if (symbols[i] == -1) {
203                     throw new GenericException(NO_SUCH_SYMBOL_OR_ATTRIBUTE, str);
204                 }
205             }
206         }
207 
208         str = symbolsOrAttributes[symbolsOrAttributes.length - 1];
209         target = spec.lookup(str);
210         if (target == null) {
211             throw new GenericException(NO_SUCH_ATTRIBUTE, str);
212         }
213         Path.translate(spec.getSyntax(), source, moves, symbols, target, Path.ISOLATED);
214         return null; // TODO
215     }

285     public static Rule[] scannerRules(List rawKeywords, Rule[] rules) throws GenericException {
286         List<Rule> keywords;
287         IntBitSet doneSymbols;
288         int i;
289         int max;
290         Rule[] all;
291         Object[] pair;
292         int symbol;
293         StringArrayList doneKeywords;
294         String kw;
295 
296         doneSymbols = new IntBitSet();
297         doneKeywords = new StringArrayList();
298         max = rawKeywords.size();
299         keywords = new ArrayList<Rule>();
300         for (i = 0; i < max; i++) {
301             pair = (Object[]) rawKeywords.get(i);
302             symbol = (Integer) pair[0];
303             if (!doneSymbols.contains(symbol)) {
304                 doneSymbols.add(symbol);
305                 kw = (String) pair[1];
306                 if (doneKeywords.indexOf(kw) != -1) {
307                     throw new GenericException("ambiguous String literal: " + kw);
308                 }
309                 keywords.add(new Rule(symbol, Sequence.createKeyword(kw)));
310             }
311         }
312         all = new Rule[keywords.size() + rules.length];
313         keywords.toArray(all);
314         System.arraycopy(rules, 0,  all, keywords.size(),  rules.length);
315         return all;
316     }
