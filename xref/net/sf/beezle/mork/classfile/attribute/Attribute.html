18  package net.sf.beezle.mork.classfile.attribute;
19  
20  import net.sf.beezle.mork.classfile.Code;
21  import net.sf.beezle.mork.classfile.Input;
22  import net.sf.beezle.mork.classfile.Output;
23  
24  import java.io.IOException;
25  
26  public abstract class Attribute {
27      public final String name;
28  
29      protected Attribute(String nameInit) {
30          name = nameInit;
31      }
32  
33      public static Attribute create(Input src) throws IOException {
34          String name;
35  
36          name = src.readUtf8();
37          if (name.equals("Code")) {
38              return new Code(src);
39          } else if (name.equals("ConstantValue")) {
40              return new ConstantValue(src);
41          } else if (name.equals("Exceptions")) {
42              return new Exceptions(src);
43          } else if (name.equals("InnerClasses")) {
44              return new InnerClasses(src);
45          } else if (name.equals("Synthetic")) {
46              return new Synthetic(src);
47          } else if (name.equals("SourceFile")) {
48              return new SourceFile(src);
49          } else if (name.equals("LineNumberTable")) {
50              return new LineNumberTable(src);
51          } else if (name.equals("LocalVariableTable")) {
52              return new LocalVariableTable(src);
53          } else if (name.equals("Deprecated")) {
54              return new Deprecated(src);
55          } else {
56              return new Blackbox(name, src);
57          }
58      }
59  
60      public abstract void write(Output dest) throws IOException;
61  }
