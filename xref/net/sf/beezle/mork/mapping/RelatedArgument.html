22  import java.util.ArrayList;
23  import java.util.List;

25  /**
26   * To sort arguments.
27   */
28  public class RelatedArgument implements Compare {
29      /**
30       * Returns the sorted list of lists of arguments.
31       *
32       * @param args   normal argument, not related arguments
33       */
34      public static List<List<Argument>> sort(List<Argument> args) {
35          int i;
36          int max;
37          List<List<Argument>> result;
38          List<RelatedArgument> remaining;    // List of RelatedArguments
39          List<RelatedArgument> heads;   // List of RelatedArguments
40  
41          result = new ArrayList<List<Argument>>();
42          remaining = createRelatedArgs(args);
43          while (true) {
44              max = remaining.size();
45              if (max == 0) {
46                  return result;
47              }
48              heads = new ArrayList<RelatedArgument>();
49              for (i = 0; i < max; i++) {
50                  addHead(heads, remaining.get(i));
51              }
52              remaining.clear();
53              result.add(separate(heads, remaining));
54          }
55      }
56  
57      /**
58       * @param args  List of Arguments
59       * @return List of RelatedArguments
60       */
61      private static List<RelatedArgument> createRelatedArgs(List<Argument> args) {
62          List<RelatedArgument> related;
63          int i;
64          int max;
65  
66          related = new ArrayList<RelatedArgument>();
67          max = args.size();
68          for (i = 0; i < max; i++) {
69              related.add(new RelatedArgument(args.get(i)));
70          }
71          return related;
72      }
73  
74      /**
75       * @param heads   List of RelatedArguments
76       */
77      private static void addHead(List<RelatedArgument> heads, RelatedArgument left) {
78          int i;
79          int max;
80          RelatedArgument right;
81          boolean foundLT;
82          boolean foundGT;
83  
84          max = heads.size();
85          foundLT = false;
86          foundGT = false;
87          for (i = max - 1; i >= 0; i--) {
88              right = heads.get(i);
89              switch (left.arg.compare(right.arg)) {
90                  case LT:
91                      left.nexts.add(right);
92                      heads.remove(i);
93                      foundLT = true;
94                      break;
95                  case GT:
96                      if (!foundGT) {
97                          right.nexts.add(left);
98                          foundGT = true;
99                      } else {
100                         // ignore - left must not be duplicated in right.nexts
101                     }
102                     break;
103                 default:
104                     // do nothing
105                     break;
106             }
107         }
108         if (foundGT && foundLT) {
109             throw new IllegalStateException();
110         }
111         if (!foundGT) {
112             heads.add(left);
113         }
114     }

116     /**
117      ** @param heads  List of RelatedArguments
118      ** @param tails out-argument, List of RelatedArguments
119      ** @return List of arguments
120      **/
121     private static List<Argument> separate(List<RelatedArgument> heads, List<RelatedArgument> tails) {
122         List<Argument> merge;
123 
124         merge = new ArrayList<Argument>();
125         for (RelatedArgument head : heads) {
126             merge.add(head.arg);
127             tails.addAll(head.nexts);
128         }
129         return merge;
130     }
