22  /** Element in a prefix set, and an iterator. Immutable, heavily shared between PrefixSets. */
23  public class Prefix {
24      public static final int BASE = 512;
25  
26      public static long pack(int ... symbols) {
27          long data;
28  
29          data = 0;
30          for (int symbol : symbols) {
31              data = data * BASE + pack(symbol);
32          }
33          return data;
34      }
35  
36      public static long pack(int symbol) {
37          if (symbol >= BASE - 1) {
38              throw new IllegalArgumentException("" + symbol);
39          }
40          return symbol + 1;
41      }
42  
43      public static int[] unpack(long data) {
44          int[] result;
45  
46          result = new int[size(data)];
47          for (int i = result.length - 1; i >= 0; i--) {
48              result[i] = (int) (data % BASE) - 1;
49              data /= BASE;
50          }
51          return result;
52      }
53  
54      //--
55  
56      /** behind current prefix - first index to check for next element */
57      private int index;
58      private final long[] table;
59  
60      /** access only for PrefixSet */
61      long data;

63      //--
64  
65      /** Constructor for PrefixSet only */
66      Prefix(long[] table, int size) {
67          this.table = table;
68          this.index = 0; // we don't have empty tables
69      }

71      //-- iterator
72  
73      public boolean step() {
74          for (; index < table.length; index++) {
75              if (table[index] != PrefixSet.FREE) {
76                  data = table[index++];
77                  return true;
78              }
79          }
80          return false;
81      }

83      //-- prefix methods
84  
85      public int first() {
86          long remaining;
87          long next;
88  
89          remaining = data;
90          while (true) {
91              next = remaining / BASE;
92              if (next == 0) {
93                  return (int) remaining - 1;
94              }
95              remaining = next;
96          }
97      }

99      public static long concat(long leftPrefix, long rightPrefix, int k) {
100         int leftSize;
101         int rightSize;
102         long newData;
103         int newSize;
104         int count;
105 
106         if (leftPrefix == 0) {
107             return rightPrefix;
108         }
109         if (rightPrefix == 0) {
110             return leftPrefix;
111         }
112         leftSize = size(leftPrefix);
113         if (leftSize == k) {
114             return leftPrefix;
115         }
116         rightSize = Prefix.size(rightPrefix);
117         newData = leftPrefix;
118         newSize = Math.min(k, leftSize + rightSize);
119         count = newSize - leftSize;
120         for (int i = count; i > 0; i--) {
121             newData *= BASE;
122         }
123         long tmp = rightPrefix;
124         for (int i = rightSize - count; i > 0; i--) {
125             tmp /= BASE;
126         }
127         return newData + tmp;
128     }

130     public int size() {
131         return size(data);
132     }
133 
134     public static int size(long prefix) {
135         int size;
136 
137         for (size = 0; prefix != 0; prefix /= BASE) {
138             size++;
139         }
140         return size;
141     }

143     public int[] follows(int first) {
144         int[] terminals;
145         int[] symbols;
146 
147         symbols = unpack(data);
148         if (symbols.length > 0) {
149             if (symbols[0] == first) {
150                 terminals = new int[symbols.length - 1];
151                 for (int i = 0; i < terminals.length; i++) {
152                     terminals[i] = symbols[(i + 1)];
153                 }
154                 return terminals;
155             }
156         }
157         return null;
158     }

160     @Override
161     public String toString() {
162         StringBuilder builder;
163 
164         builder = new StringBuilder();
165         for (int symbol : unpack(data)) {
166             builder.append(' ');
167             builder.append(symbol);
168         }
169         return builder.toString();
170     }

172     public void toString(StringArrayList symbolTable, StringBuilder result) {
173         boolean first;
174 
175         first = true;
176         for (int symbol : unpack(data)) {
177             if (first) {
178                 first = false;
179             } else {
180                 result.append(' ');
181             }
182             result.append(symbolTable.getOrIndex(symbol));
183         }
184     }

190     public static int hashNext(long prefix, int previous, int length) {
191         return (previous + 1 + (int) (prefix % (length - 2))) % length;
192     }
