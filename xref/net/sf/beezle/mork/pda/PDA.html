35  public class PDA implements PDABuilder {
36      public static PDA create(Grammar grammar, final Map<Integer, PrefixSet> firsts, final int k, int threadCount) {
37          final PDA pda;
38          State state;
39          final Queue todo;
40          int end;
41          Thread[] threads;
42          final List<Throwable> exceptions;
43  
44          threads = new Thread[threadCount];
45          todo = new Queue(threads.length);
46          state = State.forStartSymbol(grammar, grammar.getSymbolCount());
47          state.closure(grammar, firsts, k);
48          pda = new PDA(grammar, state);
49          todo.put(state);
50          exceptions = new ArrayList<Throwable>();
51          for (int i = 0; i < threads.length; i++) {
52              threads[i] = new Thread("pda-builder-" + i) {
53                  public void run() {
54                      net.sf.beezle.mork.pda.State state;
55  
56                      try {
57                          while (true) {
58                              state = todo.take();
59                              state.gotos(pda, firsts, todo, k);
60                          }
61                      } catch (InterruptedException e) {
62                          return; // terminate
63                      } catch (Throwable e) {
64                          synchronized (exceptions) {
65                              exceptions.add(e);
66                          }
67                      }
68                  }
69              };
70              threads[i].start();
71          }
72          for (Thread thread : threads) {
73              try {
74                  thread.join();
75              } catch (InterruptedException e) {
76                  throw new IllegalStateException(e);
77              }
78          }
79  
80          if (exceptions.size() > 0) {
81              throw new IllegalStateException("thread exceptions: " + exceptions.size(), exceptions.get(0));
82          }
83          // TODO: hack hack hack
84          end = pda.add(new State());
85          pda.start.shifts.add(new Shift(grammar.getStart(), end));
86          return pda;
87      }

93      public PDA(Grammar grammar, State start) {

117     public synchronized int addIfNew(State state) {

135     public ParserTable createTable(int lastSymbol, ConflictHandler handler) throws GenericException {
136         // the initial syntax node created by the start action is ignored!
137         ParserTable result;
138         int eof;
139         int end;
140 
141         eof = getEofSymbol();
142         result = new ParserTable(0, size(), lastSymbol + 1 /* +1 for EOF */, eof, grammar, null);
143         for (Map.Entry<State, Integer> entry : states.entrySet()) {
144             entry.getKey().addActions(entry.getValue(), result, handler);
145         }
146         end = start.lookupShift(grammar.getStart()).end;
147         result.addAccept(end, eof);
148         return result;
149     }

157     public void statistics(PrintStream output) {
158         int size;
159         int itemsCount;
160         int itemsMin;
161         int itemsMax;
162         int lookaheadSizes;
163         int lookaheadMin;
164         int lookaheadMax;
165         double hqMax;
166         double hqMin;
167         double hqSum;
168         double loadMax;
169         double loadMin;
170         double loadSum;
171         double q;
172 
173         itemsCount = 0;
174         itemsMin = Integer.MAX_VALUE;
175         itemsMax = Integer.MIN_VALUE;
176         lookaheadMax = Integer.MIN_VALUE;
177         lookaheadMin = Integer.MAX_VALUE;
178         lookaheadSizes = 0;
179         hqSum = 0;
180         hqMax = Double.MIN_VALUE;
181         hqMin = Double.MAX_VALUE;
182         loadSum = 0;
183         loadMax = Double.MIN_VALUE;
184         loadMin = Double.MAX_VALUE;
185         for (State state : states()) {
186             size = state.items.size();
187             itemsCount += size;
188             itemsMin = Math.min(itemsMin, size);
189             itemsMax = Math.max(itemsMax, size);
190             for (Item item : state.items) {
191                 size = item.lookahead.size();
192                 lookaheadSizes += size;
193                 lookaheadMin = Math.min(lookaheadMin, size);
194                 lookaheadMax = Math.max(lookaheadMax, size);
195                 q = item.lookahead.hashQuality();
196                 hqSum += q;
197                 hqMax = Math.max(q, hqMax);
198                 hqMin = Math.min(q, hqMin);
199                 q = item.lookahead.load();
200                 loadSum += q;
201                 loadMax = Math.max(q, loadMax);
202                 loadMin = Math.min(q, loadMin);
203             }
204         }
205         output.println("parser generation statistics");
206         output.println("  states: " + states.size());
207         output.println("  items avg: " + (itemsCount / states.size()) + ", min: " + itemsMin + ", max: " + itemsMax + ")");
208         output.println("  lookahead avg: " + (lookaheadSizes / itemsCount) + ", min: " + lookaheadMin + ", max: " + lookaheadMax);
209         output.println("  hash quality avg: " + (hqSum / itemsCount) + ", min: " + hqMin + ", max: " + hqMax);
210         output.println("  hash load avg: " + (loadSum / itemsCount) + ", min: " + loadMin + ", max: " + loadMax);
211         output.println("  heap size: " + Runtime.getRuntime().totalMemory());
212         output.println("  heap used: " + (Runtime.getRuntime().totalMemory() - Runtime.getRuntime().freeMemory()));
213     }
